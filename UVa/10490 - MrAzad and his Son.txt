ll fastpower(ll num, ll pow)
{
	if (pow == 0)
		return 1;
	ll ans = fastpower(num, pow / 2);
	ans = ans * ans;
	if (pow & 1)
		ans *= num;
	return ans;
}
bool isprime(ll num)
{
	if (num == 2)
		return 1;
	if (num < 2 || num % 2 == 0)
		return 0;
	for (ll i = 3; i*i <= num; i += 2)
		if (num%i == 0)
			return 0;
	return 1;
}
