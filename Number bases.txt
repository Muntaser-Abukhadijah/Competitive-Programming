string all = "0123456789ABCDEF";
string DtoA(int num, int base)
{
	string ans = "";
	while (num)
	{
		ans = all[num%base] + ans;
		num /= base;
	}
	return ans;
}
int AtoD(string s, int base)
{
	int ans = 0;
	for (int i = 0; i < s.size(); i++)
	{
		ans *= base;
		ans += all.find(s[i]);
	}
	return ans;
}
